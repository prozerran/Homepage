proc main

; Enable the following to delay for 3 seconds first to
; allow host time to send initial characters     
    delay 3

; Wait for the host prompt before entering
; the host name
    waitfor "ost:" 
    transmit "hills" 
    transmit "^M" 

; Wait for the login prompt before entering
; the user ID
    waitfor "ogin:"
    transmit $USERID
    transmit "^M"

; Enter the password
    waitfor "assword:"
    transmit $PASSWORD
    transmit "^M"

    delay 1

    transmit "^M"
    transmit "^M"
    transmit "^M"
    transmit "^M"

; Enter slirp command at the shell prompt:
    waitfor "$" 
    transmit "slirp ppp"
    transmit "^M"

endproc
