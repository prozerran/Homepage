@REM BATCH FILE
@ECHO off

rem =========================================================
rem Declare global variables here.
rem =========================================================

set MYSQL_DUMP_EXE="C:\Program Files\MySQL\MySQL Server 5.7\bin"
set MYSQL_USER="root"
set MYSQL_PASS="password"
set MYSQL_DB="dbname"
set DT=%date:~7,2%-%date:~4,2%-%date:~10,4%_%time:~0,2%_%time:~3,2%_%time:~6,2%
set MYSQL_OUTF="C:\Temp\DB_Backup_%DT%.sql"

rem =========================================================
rem Backup DB
rem =========================================================

%MYSQL_DUMP_EXE%\mysqldump.exe -u %MYSQL_USER% -p%MYSQL_PASS% %MYSQL_DB% > %MYSQL_OUTF%

echo "MySQL backup Done."
